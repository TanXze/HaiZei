#include <stdio.h>

#define MAX 10000

int main() {
    char a;
    char buffer[MAX];
    int word_sum = 0, n = 0;
    while (scanf("%c", &a) != EOF) {
        buffer[n++] = a;
    }
    n--;
    while (buffer[n] == ' ' || buffer[n] == '\n' && n >= 0) {
        n--;
    }
    while (buffer[n] != ' ' && buffer[n] != '\n' && n >= 0) {
        word_sum++;
        n--;
    }
    printf("%d\n", word_sum);
    return 0;
}
